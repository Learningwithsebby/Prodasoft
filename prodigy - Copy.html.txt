                this.student.name = username;

                this.closeCallback = callback;
                var flowURL     = SW.Utility.getBaseUrl() + "/membership/index.php?game=1&tag="+tag,

            },

            close: function(success){
                this.$el.hide().html("");
                this.$game.show();
                if(!_.isUndefined(this.closeCallback)) {
                    this.closeCallback(success);
                }
            }
        };

        prodigyWeb.page = {
            $el: $("#game-page-overlay"),
            $game: $("#game-wrapper"),
            closeCallback: undefined,

            getUrlForPage: function(page, studentID) {
                switch(page) {
                    case "buy-toy":
                        var version = "";
                        
                        if(studentID % 2 == 0) version = "vid-embed";
                        else version = "no-vid";

                        return "/toys/hex.php?referrer=game&version="+version;
                        break;
                    case "unlock-toy":
                        return "/toys/unlock.php?referrer=game";
                        break;
                }

                return "";
            },

            open: function(page, data, closeCallback){
                this.data = data;
                this.closeCallback = closeCallback;


                var flowURL     = SW.Utility.getBaseUrl() + this.getUrlForPage(page, data.studentID),
                    $iframeFlow = $("<iframe src='" + flowURL + "'>");
                
                this.$el.show()
                        .append($iframeFlow);
                this.$game.hide();

                this.$el.find("iframe").focus();
            },

            close: function(success){
                this.$el.hide().html("");
                this.$game.show();
                this.$game.focus();
                if(!_.isUndefined(this.closeCallback)) {
                    this.closeCallback(success);
                }
            }
        }
